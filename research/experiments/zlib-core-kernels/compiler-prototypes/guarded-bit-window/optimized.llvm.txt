bw_tail_guard:
  %bw_tail_byte_within = icmp ule i64 %bw_tail_byte, %bw_src_len
  %bw_tail_src_available = sub i64 %bw_src_len, %bw_tail_byte
  %bw_tail_has_two = icmp uge i64 %bw_tail_src_available, 2
  %bw_tail_input_guard = and i1 %bw_tail_byte_within, %bw_tail_has_two
  %bw_tail_output_within = icmp ult i64 %bw_tail_produced, %bw_out_len
  %bw_tail_guard_ok = and i1 %bw_tail_input_guard, %bw_tail_output_within
  br i1 %bw_tail_guard_ok, label %bw_tail_load, label %trap
bw_tail_load:
  %bw_tail_ptr0 = getelementptr i8, ptr %bw_src_ptr, i64 %bw_tail_byte
  %bw_tail_byte0 = load i8, ptr %bw_tail_ptr0, align 1
  %bw_tail_byte1_index = add i64 %bw_tail_byte, 1
  %bw_tail_ptr1 = getelementptr i8, ptr %bw_src_ptr, i64 %bw_tail_byte1_index
  %bw_tail_byte1 = load i8, ptr %bw_tail_ptr1, align 1
  %bw_tail_wide0 = zext i8 %bw_tail_byte0 to i32
  %bw_tail_wide1 = zext i8 %bw_tail_byte1 to i32
  %bw_tail_shifted1 = shl i32 %bw_tail_wide1, 8
  %bw_tail_window = or i32 %bw_tail_wide0, %bw_tail_shifted1
  %bw_tail_offset32 = trunc i64 %bw_tail_offset to i32
  %bw_tail_hold = lshr i32 %bw_tail_window, %bw_tail_offset32
  %bw_tail_index32 = and i32 %bw_tail_hold, 511
  %bw_tail_index = zext i32 %bw_tail_index32 to i64
  %bw_tail_entry_ptr = getelementptr [512 x i32], ptr @__const_fixed_lencode, i64 0, i64 %bw_tail_index
  %bw_tail_entry_value = load i32, ptr %bw_tail_entry_ptr, align 4
  %bw_tail_operation = and i32 %bw_tail_entry_value, 255
  %bw_tail_is_literal = icmp eq i32 %bw_tail_operation, 0
  br i1 %bw_tail_is_literal, label %bw_tail_decode, label %trap
bw_tail_decode:
  %bw_tail_symbol_word = lshr i32 %bw_tail_entry_value, 16
  %bw_tail_symbol = trunc i32 %bw_tail_symbol_word to i8
  %bw_tail_out_at = getelementptr i8, ptr %bw_out_ptr, i64 %bw_tail_produced
  store i8 %bw_tail_symbol, ptr %bw_tail_out_at, align 1
  %bw_tail_used_shifted = lshr i32 %bw_tail_entry_value, 8
  %bw_tail_used32 = and i32 %bw_tail_used_shifted, 255
  %bw_tail_used = zext i32 %bw_tail_used32 to i64
  %bw_tail_sum = add i64 %bw_tail_offset, %bw_tail_used
  %bw_tail_advance = lshr i64 %bw_tail_sum, 3
  %bw_tail_next_offset = and i64 %bw_tail_sum, 7
  %bw_tail_next_byte = add i64 %bw_tail_byte, %bw_tail_advance
  %bw_tail_next_produced = add i64 %bw_tail_produced, 1
  br label %bw_tail_header
bw_return:
  ret i64 %bw_tail_produced
trap:
  call void @llvm.trap()
  unreachable
}
